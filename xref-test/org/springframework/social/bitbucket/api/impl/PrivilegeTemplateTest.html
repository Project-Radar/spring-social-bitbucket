25  import static org.junit.Assert.assertEquals;
26  import static org.springframework.http.HttpMethod.*;
27  import static org.springframework.test.web.client.match.MockRestRequestMatchers.*;
28  import static org.springframework.test.web.client.response.MockRestResponseCreators.withNoContent;
29  import static org.springframework.test.web.client.response.MockRestResponseCreators.withSuccess;

31  public class PrivilegeTemplateTest extends BaseTemplateTest {
32      @Test
33      public void testGetRepoPrivileges() {
34  
35          mockServer
36                  .expect(requestTo("https://api.bitbucket.org/1.0/privileges/evzijst/test"))
37                  .andExpect(method(GET))
38                  .andRespond(
39                          withSuccess(jsonResource("get-repo-privileges"),
40                                  MediaType.APPLICATION_JSON));
41  
42          List<RepoPrivilege> privs = bitBucket.privelegesOperations()
43                  .getRepoPrivileges("evzijst", "test");
44  
45          assertEquals(4, privs.size());
46          assertEquals("evzijst/test", privs.get(0).getRepository());
47          assertEquals("jespern", privs.get(0).getUser().getUsername());
48          assertEquals(BitBucketPrivilege.read, privs.get(0).getPrivilege());
49  
50      }
51  
52      @Test
53      public void testDeletePrivilege() {
54          mockServer
55                  .expect(requestTo("https://api.bitbucket.org/1.0/privileges/evzijst/test/jespern"))
56                  .andExpect(method(DELETE)).andRespond(withNoContent());
57  
58          bitBucket.privelegesOperations().removePrivilege("evzijst", "test",
59                  "jespern");
60      }
61  
62      @Test
63      public void testSetPrivilege() {
64          mockServer
65                  .expect(requestTo("https://api.bitbucket.org/1.0/privileges/tutorials/mydvcsproject/atlassian_tutorial"))
66                  .andExpect(method(PUT))
67                  .andExpect(content().string("write"))
68                  .andRespond(
69                          withSuccess(jsonResource("set-repo-privilege"),
70                                  MediaType.APPLICATION_JSON));
71  
72          RepoPrivilege priv = bitBucket.privelegesOperations().setPrivilege(
73                  "tutorials", "mydvcsproject", "atlassian_tutorial",
74                  BitBucketPrivilege.write);
75  
76          assertEquals("Atlassian Tutorials", priv.getUser().getFirstName());
77      }
