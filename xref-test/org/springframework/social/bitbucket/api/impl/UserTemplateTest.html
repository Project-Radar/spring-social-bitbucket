36      @Test
37      public void testGetUser() throws Exception {
38  
39          mockServer
40                  .expect(requestTo("https://api.bitbucket.org/1.0/user"))
41                  .andExpect(method(GET))
42                  .andRespond(
43                          withSuccess(jsonResource("get-user"),
44                                  MediaType.APPLICATION_JSON));
45  
46          UserWithRepositories profile = bitBucket.userOperations()
47                  .getUserWithRepositories();
48  
49          assertEquals("ericbottard", profile.getUser().getUsername());
50          assertEquals("Eric", profile.getUser().getFirstName());
51          assertEquals("Bottard", profile.getUser().getLastName());
52          assertEquals(
53                  "https://secure.gravatar.com/avatar/9adf9e4ee93fea30ea27b469cfa3bae8?d=identicon&s=32",
54                  profile.getUser().getAvatarImageUrl());
55  
56          assertEquals(3, profile.getRepositories().size());
57          BitBucketRepository repo = profile.getRepositories().iterator().next();
58  
59          SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
60          sdf.setTimeZone(TimeZone.getTimeZone("GMT"));
61  
62          assertEquals(sdf.parse("2011-11-20 18:25:29"), repo.getCreatedAt());
63          assertEquals("ericbottard", repo.getOwner());
64          assertEquals(BitBucketSCM.git, repo.getScm());
65          assertTrue(repo.isHasWiki());
66          assertEquals(29397517L, repo.getSize());
67          assertFalse(repo.isReadOnly());
68          assertTrue(repo.isPrivate());
69          assertEquals(sdf.parse("2012-02-20 21:29:16+00:00"),
70                  repo.getLastUpdatedOn());
71  
72      }
