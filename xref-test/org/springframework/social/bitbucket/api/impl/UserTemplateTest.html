25  import java.text.SimpleDateFormat;
26  import java.util.List;

28  
29  import static org.junit.Assert.*;
30  import static org.springframework.http.HttpMethod.GET;
31  import static org.springframework.test.web.client.match.MockRestRequestMatchers.method;
32  import static org.springframework.test.web.client.match.MockRestRequestMatchers.requestTo;
33  import static org.springframework.test.web.client.response.MockRestResponseCreators.withSuccess;
34  
35  public class UserTemplateTest extends BaseTemplateTest {
36  
37      @Test
38      public void testGetUser() throws Exception {
39  
40          mockServer
41                  .expect(requestTo("https://api.bitbucket.org/1.0/user"))
42                  .andExpect(method(GET))
43                  .andRespond(
44                          withSuccess(jsonResource("get-user"),
45                                  MediaType.APPLICATION_JSON));
46  
47          UserWithRepositories profile = bitBucket.userOperations()
48                  .getUserWithRepositories();
49  
50          assertEquals("ericbottard", profile.getUser().getUsername());
51          assertEquals("Eric", profile.getUser().getFirstName());
52          assertEquals("Bottard", profile.getUser().getLastName());
53          assertEquals(
54                  "https://secure.gravatar.com/avatar/9adf9e4ee93fea30ea27b469cfa3bae8?d=identicon&s=32",
55                  profile.getUser().getAvatarImageUrl());
56  
57          assertEquals(3, profile.getRepositories().size());
58          BitBucketRepository repo = profile.getRepositories().iterator().next();
59  
60          SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
61          sdf.setTimeZone(TimeZone.getTimeZone("GMT"));
62  
63          assertEquals(sdf.parse("2011-11-20 18:25:29"), repo.getCreatedAt());
64          assertEquals("ericbottard", repo.getOwner());
65          assertEquals(BitBucketSCM.git, repo.getScm());
66          assertTrue(repo.isHasWiki());
67          assertEquals(29397517L, repo.getSize());
68          assertFalse(repo.isReadOnly());
69          assertTrue(repo.isPrivate());
70          assertEquals(sdf.parse("2012-02-20 21:29:16+00:00"),
71                  repo.getLastUpdatedOn());
72  
73      }
74  
75      @Test
76      public void testGetUserFollowers() throws Exception {
77  
78          mockServer
79                  .expect(requestTo("https://api.bitbucket.org/1.0/users/cleonello/followers"))
80                  .andExpect(method(GET))
81                  .andRespond(
82                          withSuccess(jsonResource("user-followers"),
83                                  MediaType.APPLICATION_JSON));
84  
85          List<BitBucketUser> followers = bitBucket.userOperations()
86                  .getFollowers("cleonello");
87          assertEquals(11, followers.size());
88          BitBucketUser follower = followers.get(0);
89  
90          assertEquals("tobami", follower.getUsername());
91          assertEquals("Miquel", follower.getFirstName());
92          assertEquals("Torres", follower.getLastName());
93          assertEquals(
94                  "https://secure.gravatar.com/avatar/a804ab4bb149c5c612a087e75d2c656c?d=identicon&s=32",
95                  follower.getAvatarImageUrl());
96      }
