16  package org.springframework.social.bitbucket.api.v2;
17  
18  import org.springframework.social.bitbucket.api.v2.payload.*;
19  
20  import java.util.Set;
21  
22  public interface PullRequestV2Operations {
23  
24      /**
25       * Get a list of of a repository's open pull requests.
26       * <p/>
27       * This returns a paginated list of pull request objects. Note that the participants list is not included in the
28       * pull requests. To obtain the full list of participants, follow the pull requests's self link.
29       * <p/>
30       * By default, only the open pull requests are returned. Use the state query parameter to access declined
31       * (state=DECLINED), open (state=OPEN) or merged (state=MERGED). The state parameter can be repeated to include pull
32       * requests in different states.
33       *
34       * @param owner    The account owning the repo.
35       * @param repoSlug The repo identifier.
36       * @param states   Filter the pull requests on those states. If null or empty, only open pull requests are returned.
37       */
38      PullRequestPage getPullRequests(String owner, String repoSlug, Set<BitBucketV2PullRequestState> states);
39  
40      /**
41       * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#getPullRequests(String, String, java.util.Set)
42       */
43      PullRequestPage getPullRequests(BitBucketV2Account owner, String repoSlug, Set<BitBucketV2PullRequestState> states);
44  
45      /**
46       * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#getPullRequests(String, String, java.util.Set)
47       */
48      PullRequestPage getPullRequests(BitBucketV2Repository repository, Set<BitBucketV2PullRequestState> states);
49  
50      /**
51       * Creates a new pull request. The request URL you provide is the destination repository URL. For this reason, you
52       * must specify an explicit source repository in the request object if you want to pull from a different repository.
53       *
54       * @param owner    Owner of the destination repository.
55       * @param repoSlug The repo identifier of the destination repository.
56       */
57      BitBucketV2PullRequest createPullRequest(
58              String owner,
59              String repoSlug,
60              BitBucketV2PullRequest pullRequest);
61  
62      /**
63       * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#createPullRequest(String, String, org.springframework.social.bitbucket.api.v2.payload.BitBucketV2PullRequest)
64       */
65      BitBucketV2PullRequest createPullRequest(
66              BitBucketV2Account owner,
67              String repoSlug,
68              BitBucketV2PullRequest pullRequest);
69  
70      /**
71       * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#createPullRequest(String, String, org.springframework.social.bitbucket.api.v2.payload.BitBucketV2PullRequest)
72       */
73      BitBucketV2PullRequest createPullRequest(
74              BitBucketV2Repository repository,
75              BitBucketV2PullRequest pullRequest);
76  
77      /**
78       * Updates an existing pull request.
79       * <p/>
80       * The pull request's status must be open. With the exception of the source and destination parameters, the request
81       * body must include all the existing request parameters; Omitting a parameter causes the server to drop the
82       * existing value. For example, if the pull requests already has 3 reviewers, the request body must include these 3
83       * reviewers to prevent Bitbucket from dropping them.
84       * <p/>
85       * A pull request cannot exist without the source and destination parameters. For this reason Bitbucket never drops
86       * them. The URL provides the destination and Bitbucket keeps the existing source.
87       * <p/>
88       * A caller cannot change any elements of the source repository or the destination repository's name. A caller can
89       * change the destination branch. To do this, provide the destination's branch element in the call.
90       * <p/>
91       * A pull request is stale when new commits were pushed to the source branch but the request does not yet reflect
92       * these new commits. When a PUT is called on a stale request, Bitbucket always updates the source branch to the new
93       * head commit before processing the PUT.
94       */
95      BitBucketV2PullRequest updatePullRequest(
96              String owner, String repoSlug, String id, BitBucketV2PullRequest pullRequest);
97  
98      /**
99       * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#updatePullRequest(String, String, String, org.springframework.social.bitbucket.api.v2.payload.BitBucketV2PullRequest)
100      */
101     BitBucketV2PullRequest updatePullRequest(
102             BitBucketV2Account owner, String repoSlug, String id, BitBucketV2PullRequest pullRequest);
103 
104     /**
105      * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#updatePullRequest(String, String, String, org.springframework.social.bitbucket.api.v2.payload.BitBucketV2PullRequest)
106      */
107     BitBucketV2PullRequest updatePullRequest(
108             BitBucketV2Repository repository, String id, BitBucketV2PullRequest pullRequest);
109 
110     /**
111      * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#updatePullRequest(String, String, String, org.springframework.social.bitbucket.api.v2.payload.BitBucketV2PullRequest)
112      */
113     BitBucketV2PullRequest updatePullRequest(
114             BitBucketV2PullRequest existingPullRequest, BitBucketV2PullRequest updatedPullRequest);
115 
116     /**
117      * Gets an specific pull request.
118      *
119      * @param owner    The account owning the repo.
120      * @param repoSlug The repo identifier.
121      * @param id       The pull request identifier.
122      */
123     BitBucketV2PullRequest getPullRequest(String owner, String repoSlug, String id);
124 
125     /**
126      * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#getPullRequest(String, String, String)
127      */
128     BitBucketV2PullRequest getPullRequest(BitBucketV2Account owner, String repoSlug, String id);
129 
130     /**
131      * @see org.springframework.social.bitbucket.api.v2.PullRequestV2Operations#getPullRequest(String, String, String)
132      */
133     BitBucketV2PullRequest getPullRequest(BitBucketV2Repository repository, String id);
