16  package org.springframework.social.bitbucket.api;
17  
18  import java.util.List;
19  import java.util.Map;
20  
21  /**
22   * BitBucket sub-API that allows interaction with code repositories.
23   *
24   * @author ericbottard
25   */
26  public interface RepoOperations {
27  
28      /**
29       * Returns the public repository that belongs to {@code user} and that has
30       * {@code repoSlug} as its technical name.
31       */
32      BitBucketRepository getRepository(String user, String repoSlug);
33  
34      /**
35       * Returns the list of repositories for the current user.
36       */
37      List<BitBucketRepository> getUserRepositories();
38  
39      /**
40       * Search for public repositories whose name contains the given String.
41       */
42      List<BitBucketRepository> search(String string);
43  
44      /**
45       * Returns a mapping between tags (keys) and changesets (values) on the
46       * given repository.
47       */
48      Map<String, BitBucketChangeset> getTags(String user, String repoSlug);
49  
50      /**
51       * Returns the list of users following the given repository.
52       */
53      List<BitBucketUser> getFollowers(String user, String repoSlug);
54  
55      /**
56       * Returns the last changesets on a repository, as well as information about
57       * how many are available totally.
58       */
59      BitBucketChangesets getChangesets(String user, String repoSlug);
60  
61      /**
62       * Returns some changesets on a repository, as well as information about how
63       * many are available totally.
64       */
65      BitBucketChangesets getChangesets(String user, String repoSlug,
66                                        String start, int limit);
67  
68      /**
69       * Returns information about a known directory, including children
70       * directories and files.
71       */
72      BitBucketDirectory getDirectory(String user, String repoSlug,
73                                      String revision, String path);
74  
75      /**
76       * Returns information and actual contents (as a String) about a known file
77       * path.
78       */
79      BitBucketFile getFile(String string, String string2, String string3,
80                            String string4);
81  
82      /**
83       * Creates a new repository under the account of the currently authenticated
84       * user. The account automatically becomes the owner.
85       */
86      BitBucketRepository createRepository(RepoCreation options);
87  }
